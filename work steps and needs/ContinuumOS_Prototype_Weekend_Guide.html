# api/main.py
from fastapi import FastAPI, UploadFile, File
from fastapi.middleware.cors import CORSMiddleware
import shutil

app = FastAPI(title="ContinuumOS Prototype")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

@app.post("/upload")
async def upload_heritage(
    photo: UploadFile = File(...),
    text: UploadFile = File(...)
):
    # Save files
    with open(f"uploads/{photo.filename}", "wb") as f:
        shutil.copyfileobj(photo.file, f)
    
    with open(f"uploads/{text.filename}", "wb") as f:
        shutil.copyfileobj(text.file, f)
    
    return {"message": "Files uploaded", "photo": photo.filename}

# services/memory_store.py
import pinecone
from sentence_transformers import SentenceTransformer

pinecone.init(api_key="YOUR-API-KEY", environment="gcp-starter")

# Create index if not exists
if "continuum-demo" not in pinecone.list_indexes():
    pinecone.create_index("continuum-demo", dimension=384, metric="cosine")

index = pinecone.Index("continuum-demo")
model = SentenceTransformer('all-MiniLM-L6-v2')

def store_memory(ancestor_id: str, text: str):
    embedding = model.encode(text).tolist()
    index.upsert([(f"{ancestor_id}-{hash(text)}", embedding, {"text": text})])
    
# Test it
store_memory("grandpa-joe", "I loved fishing at the lake every Sunday")

# services/persona.py
import openai
from .memory_store import index, model

openai.api_key = "YOUR-OPENAI-KEY"

def create_persona(name: str, bio: str, memories: list):
    return f"""You are {name}. {bio}
    
Your memories include:
{chr(10).join(f"- {m}" for m in memories)}

Respond as {name} would, using their speech patterns and knowledge.
Never break character. If asked about something you don't know, 
respond as {name} would have."""

def chat_with_ancestor(ancestor_id: str, message: str):
    # Search memories
    query_embedding = model.encode(message).tolist()
    results = index.query(
        vector=query_embedding,
        top_k=5,
        include_metadata=True,
        filter={"ancestor_id": ancestor_id}
    )
    
    memories = [r.metadata['text'] for r in results.matches]
    
    # Generate response
    response = openai.ChatCompletion.create(
        model="gpt-3.5-turbo",
        messages=[
            {"role": "system", "content": create_persona("Grandpa Joe", "A fisherman", memories)},
            {"role": "user", "content": message}
        ]
    )
    
    return response.choices[0].message.content

Environment Setup

~5 minutes

Sets up Python, Node.js, and all dependencies

#!/bin/bash
# setup-environment.sh

echo "üöÄ ContinuumOS Environment Setup"
echo "================================"

# Check OS
if [[ "$OSTYPE" == "darwin"* ]]; then
    OS="macos"
elif [[ "$OSTYPE" == "linux-gnu"* ]]; then
    OS="linux"
else
    OS="windows"
fi

# Install dependencies
echo "üì¶ Installing system dependencies..."

if [ "$OS" = "macos" ]; then
    # Install Homebrew if needed
    if ! command -v brew &> /dev/null; then
        /bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
    fi
    
    brew install python@3.10 node ffmpeg tesseract
    brew install --cask docker
elif [ "$OS" = "linux" ]; then
    sudo apt update
    sudo apt install -y python3.10 python3-pip nodejs npm ffmpeg tesseract-ocr docker.io docker-compose
    sudo usermod -aG docker $USER
fi

# Create project structure
echo "üìÅ Creating project structure..."
mkdir -p continuum-os/{api,web,services,uploads,scripts}
cd continuum-os

# Python virtual environment
echo "üêç Setting up Python environment..."
python3 -m venv venv
source venv/bin/activate
pip install --upgrade pip

# Create requirements.txt
cat > requirements.txt << EOF
fastapi==0.104.1
uvicorn[standard]==0.24.0
openai==1.3.0
pinecone-client==2.2.4
langchain==0.0.340
sentence-transformers==2.2.2
neo4j==5.14.0
python-multipart==0.0.6
redis==5.0.1
boto3==1.29.7
stripe==7.4.0
elevenlabs==0.2.26
EOF

pip install -r requirements.txt

# Node.js setup
echo "üì¶ Setting up Node.js project..."
cd web
npx create-next-app@latest . --typescript --tailwind --app --no-git
npm install axios react-dropzone framer-motion socket.io-client

echo "‚úÖ Environment setup complete!"
echo "üìù Next: Run './start-services.sh' to launch"

Docker Compose Generator

~2 minutes

Creates complete Docker setup with all services

#!/bin/bash
# generate-docker-compose.sh

echo "üê≥ Generating Docker Compose configuration..."

# Create .env file
cat > .env << EOF
# API Keys (add your own)
OPENAI_API_KEY=sk-...
PINECONE_API_KEY=...
PINECONE_ENV=gcp-starter
ELEVENLABS_API_KEY=...

# Database
NEO4J_PASSWORD=continuum123
REDIS_PASSWORD=redis123

# Services
API_PORT=8000
WEB_PORT=3000
NEO4J_PORT=7474
EOF

# Create docker-compose.yml
cat > docker-compose.yml << 'EOF'
version: '3.9'

services:
  api:
    build: ./api
    ports:
      - "${API_PORT}:8000"
    environment:
      - OPENAI_API_KEY=${OPENAI_API_KEY}
      - PINECONE_API_KEY=${PINECONE_API_KEY}
      - NEO4J_URI=bolt://neo4j:7687
      - REDIS_URL=redis://redis:6379
    volumes:
      - ./uploads:/app/uploads
    depends_on:
      - neo4j
      - redis

  web:
    build: ./web
    ports:
      - "${WEB_PORT}:3000"
    environment:
      - NEXT_PUBLIC_API_URL=http://localhost:${API_PORT}
    depends_on:
      - api

  neo4j:
    image: neo4j:5-community
    ports:
      - "${NEO4J_PORT}:7474"
      - "7687:7687"
    environment:
      - NEO4J_AUTH=neo4j/${NEO4J_PASSWORD}
    volumes:
      - neo4j_data:/data

  redis:
    image: redis:7-alpine
    command: redis-server --requirepass ${REDIS_PASSWORD}
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data

volumes:
  neo4j_data:
  redis_data:
EOF

# Create API Dockerfile
cat > api/Dockerfile << 'EOF'
FROM python:3.10-slim

WORKDIR /app

COPY requirements.txt .
RUN pip install -r requirements.txt

COPY . .

CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
EOF

# Create Web Dockerfile
cat > web/Dockerfile << 'EOF'
FROM node:18-alpine

WORKDIR /app

COPY package*.json ./
RUN npm ci

COPY . .
RUN npm run build

CMD ["npm", "start"]
EOF

echo "‚úÖ Docker configuration generated!"
echo "üìù Edit .env file with your API keys"
echo "üöÄ Run 'docker-compose up' to start"

API Scaffold Generator

~1 minute

Creates complete FastAPI structure with all endpoints

#!/bin/bash
# scaffold-api.sh

echo "‚ö° Scaffolding ContinuumOS API..."

cd api

# Create directory structure
mkdir -p {routers,services,models,utils}

# Main API file
cat > main.py << 'EOF'
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from routers import heritage, chat, ancestors

app = FastAPI(title="ContinuumOS API", version="0.1.0")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

# Include routers
app.include_router(heritage.router, prefix="/heritage", tags=["heritage"])
app.include_router(chat.router, prefix="/chat", tags=["chat"])
app.include_router(ancestors.router, prefix="/ancestors", tags=["ancestors"])

@app.get("/")
def read_root():
    return {"message": "ContinuumOS API Running", "version": "0.1.0"}

@app.get("/health")
def health_check():
    return {"status": "healthy"}
EOF

# Heritage router
cat > routers/heritage.py << 'EOF'
from fastapi import APIRouter, UploadFile, File
from typing import List
import services.storage as storage

router = APIRouter()

@router.post("/upload")
async def upload_heritage(
    photos: List[UploadFile] = File(...),
    documents: List[UploadFile] = File(None),
    audio: List[UploadFile] = File(None)
):
    """Upload heritage materials for an ancestor"""
    results = await storage.process_uploads(photos, documents, audio)
    return {"status": "uploaded", "files": results}
EOF

# Chat router
cat > routers/chat.py << 'EOF'
from fastapi import APIRouter, WebSocket
from pydantic import BaseModel
import services.ai as ai

router = APIRouter()

class ChatMessage(BaseModel):
    ancestor_id: str
    message: str

@router.post("/message")
async def send_message(chat: ChatMessage):
    """Send a message to an ancestor"""
    response = await ai.generate_response(chat.ancestor_id, chat.message)
    return {"response": response}

@router.websocket("/ws/{ancestor_id}")
async def websocket_chat(websocket: WebSocket, ancestor_id: str):
    """Real-time chat with ancestor"""
    await websocket.accept()
    while True:
        data = await websocket.receive_text()
        response = await ai.generate_response(ancestor_id, data)
        await websocket.send_text(response)
EOF

# AI Service
cat > services/ai.py << 'EOF'
import openai
from services.memory import search_memories

async def generate_response(ancestor_id: str, message: str) -> str:
    """Generate AI response based on ancestor memories"""
    # Search relevant memories
    memories = await search_memories(ancestor_id, message)
    
    # Create context
    context = "\n".join([m["text"] for m in memories])
    
    # Generate response
    response = openai.ChatCompletion.create(
        model="gpt-3.5-turbo",
        messages=[
            {"role": "system", "content": f"You are {ancestor_id}. Context: {context}"},
            {"role": "user", "content": message}
        ]
    )
    
    return response.choices[0].message.content
EOF

echo "‚úÖ API scaffold complete!"
echo "üìù Add your business logic to the services"

Production Deploy

~10 minutes

Deploys to AWS with one command

#!/bin/bash
# deploy-production.sh

echo "üöÄ Deploying ContinuumOS to Production..."

# Check AWS credentials
if ! aws sts get-caller-identity > /dev/null 2>&1; then
    echo "‚ùå AWS credentials not configured"
    echo "Run: aws configure"
    exit 1
fi

# Variables
REGION="us-east-1"
CLUSTER_NAME="continuum-cluster"
SERVICE_NAME="continuum-api"

# Build and push Docker images
echo "üê≥ Building Docker images..."
docker build -t continuum-api ./api
docker build -t continuum-web ./web

# Tag and push to ECR
AWS_ACCOUNT=$(aws sts get-caller-identity --query Account --output text)
ECR_REGISTRY="$AWS_ACCOUNT.dkr.ecr.$REGION.amazonaws.com"

aws ecr get-login-password --region $REGION | docker login --username AWS --password-stdin $ECR_REGISTRY

docker tag continuum-api:latest $ECR_REGISTRY/continuum-api:latest
docker tag continuum-web:latest $ECR_REGISTRY/continuum-web:latest

docker push $ECR_REGISTRY/continuum-api:latest
docker push $ECR_REGISTRY/continuum-web:latest

# Deploy with ECS
echo "‚òÅÔ∏è Deploying to ECS..."
aws ecs update-service \
    --cluster $CLUSTER_NAME \
    --service $SERVICE_NAME \
    --force-new-deployment \
    --region $REGION

echo "‚úÖ Deployment initiated!"
echo "üìä Monitor at: https://console.aws.amazon.com/ecs"

#!/bin/bash
# master-setup.sh
# Complete ContinuumOS setup in one script

set -e  # Exit on error

echo "üåü ContinuumOS Master Setup"
echo "=========================="
echo ""

# Colors
GREEN='\033[0;32m'
BLUE='\033[0;34m'
YELLOW='\033[1;33m'
NC='\033[0m'

# Function to print colored output
print_status() {
    echo -e "${BLUE}[$(date +'%H:%M:%S')]${NC} $1"
}

print_success() {
    echo -e "${GREEN}‚úì${NC} $1"
}

print_warning() {
    echo -e "${YELLOW}‚ö†${NC} $1"
}

# Check prerequisites
print_status "Checking prerequisites..."

if ! command -v git &> /dev/null; then
    print_warning "Git not installed. Installing..."
    if [[ "$OSTYPE" == "darwin"* ]]; then
        brew install git
    else
        sudo apt-get install -y git
    fi
fi

# Clone repository (if needed)
if [ ! -d "continuum-os" ]; then
    print_status "Creating project structure..."
    mkdir -p continuum-os
    cd continuum-os
else
    cd continuum-os
fi

# Run setup scripts in order
print_status "Setting up environment..."
curl -sSL https://raw.githubusercontent.com/continuum-os/setup/main/setup-environment.sh | bash

print_status "Generating Docker configuration..."
curl -sSL https://raw.githubusercontent.com/continuum-os/setup/main/generate-docker-compose.sh | bash

print_status "Scaffolding API..."
curl -sSL https://raw.githubusercontent.com/continuum-os/setup/main/scaffold-api.sh | bash

# Check for API keys
if [ ! -f .env ] || ! grep -q "sk-" .env; then

    echo ""
    echo "Please add your API keys to the .env file:"
    echo "  - OpenAI API Key"
    echo "  - Pinecone API Key"

fi

# Wait for services
print_status "Waiting for services to start..."
sleep 10

# Run tests
print_status "Running health checks..."
curl -sSL https://raw.githubusercontent.com/continuum-os/setup/main/test-setup.sh | bash

# Seed demo data
print_status "Seeding demo data..."
python3 seed-demo-data.py

echo ""
echo "üéâ ContinuumOS setup complete!"

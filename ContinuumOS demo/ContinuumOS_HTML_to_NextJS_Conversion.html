Add API Routes

Create backend endpoints right in your Next.js app!

// app/api/ancestors/route.ts
import { NextResponse } from 'next/server'

// GET all ancestors
export async function GET() {
  // In real app, fetch from database
  const ancestors = [
    { id: 1, name: 'Grandpa Joe', born: 1925 },
    { id: 2, name: 'Grandma Mary', born: 1928 }
  ]
  
  return NextResponse.json(ancestors)
}

// POST new ancestor
export async function POST(request: Request) {
  const data = await request.json()
  
  // Validate data
  if (!data.name || !data.bio) {
    return NextResponse.json(
      { error: 'Name and bio required' },
      { status: 400 }
    )
  }
  
  // Save to database (example)
  const newAncestor = {
    id: Date.now(),
    ...data,
    createdAt: new Date()
  }
  
  return NextResponse.json(newAncestor, { status: 201 })
}
